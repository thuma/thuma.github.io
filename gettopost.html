Usage: gettopost.html#url&&&postfield1=value1&postfield2=value2
gettopost.html = Scriptname
# = Divider to the data to use for the POST request
url = the url including any GET fields.
&&& = divider for the data to be POSTed.
postfieldN = the fieldname to be POSTed.
valueN = the value to be POSTed.
